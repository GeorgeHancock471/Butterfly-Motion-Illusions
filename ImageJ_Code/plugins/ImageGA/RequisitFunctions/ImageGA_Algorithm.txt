  function index(a, value) {
      for (i=0; i<a.length; i++)
          if (a[i]==value) return i;
      return -1;
  }
  
 // FindID
//---------------------------------------------------
// Allows you to obtain the index of a Value within an Array.

  function findID(a, value) {
      for (i=0; i<a.length; i++){
		tA = split(a[i], "\t");
		tStr = tA[0];
          if (tStr == value) return i;
		  }
      return -1;
  }
